    `;

function openModal(id, e) {

  e.stopPropagation();

  const p = paths.find(x => x.id === id);

  document.getElementById('mIcon').textContent = p.icon;

  document.getElementById('mTitle').textContent = `路径 \${p.id}：${p.title}`;

  document.getElementById('mConcept').textContent = p.concept;

  document.getElementById('mCases').innerHTML = p.deepCases.map(c=>`

    <div class="modal-case">

      <strong>${c.title}</strong><br>${c.desc}

    </div>

  `).join('');

  document.getElementById('modalOverlay').classList.add('open');

}
